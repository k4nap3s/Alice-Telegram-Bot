game.py 
""" 
handlers/game.py — Game lifecycle commands (create, join, start, end, status). 
 
All handlers use reply keyboards for better UX and include input validation. 
"""

import logging
import random
from telegram import Update, ReplyKeyboardMarkup
from telegram.ext import ContextTypes

from models import GameSession, GameStatus, Player
from database import Database
from content import PLAYER_SECRETS
from config import GAME_CONFIG

logger = logging.getLogger(__name__)

# In-memory game cache (synced with DB)
games: dict[str, GameSession] = {}
db = Database()

# Load existing games on startup
games = db.load_all_active_games()


def generate_game_id() -> str:
    """Generate a short, memorable 6-character game ID."""
    while True:
        gid = "".join(random.choices(GAME_CONFIG.GAME_ID_CHARS, k=GAME_CONFIG.GAME_ID_LENGTH))
        if gid not in games:
            return gid


def get_main_keyboard():
    """Return main menu keyboard."""
    return ReplyKeyboardMarkup(
        [
            ["🎮 /newgame", "👥 /join"],
            ["▶️ /startgame", "⏹️ /endgame"],
            ["📋 /status", "❓ /help"],
        ],
        one_time_keyboard=True,
        resize_keyboard=True,
    )


async def cmd_newgame(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Create a new game session."""
    user = update.effective_user

    # Check if user already hosts an active game
    for g in games.values():
        if g.host_id == user.id and not g.is_ended():
            await update.message.reply_text(
                f"❌ You already have an active game (ID: `{g.game_id}`). "
                f"Use `/endgame` first.\n\n"
                f"To check status: `/status`",
                parse_mode="Markdown",
                reply_markup=get_main_keyboard(),
            )
            return

    session = GameSession(host_id=user.id, host_name=user.first_name)
    games[session.game_id] = session
    db.save_game(session)

    keyboard = [
        [f"📋 Share: {session.game_id}"],
        ["✅ Players joined?", "▶️ /startgame"],
        ["📋 /status", "❓ /help"],
    ]
    reply_markup = ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True)

    await update.message.reply_text(
        f"🎮 *New game created!*\n\n"
        f"Your Game ID is: `{session.game_id}`\n\n"
        f"Share this ID with other players so they can join:\n`/join {session.game_id}`\n\n"
        f"When everyone is ready, use `/startgame` to begin.",
        parse_mode="Markdown",
        reply_markup=reply_markup,
    )
    logger.info("New game created: %s by %s (%d)", session.game_id, user.first_name, user.id)


async def cmd_join(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Join an existing game session."""
    user = update.effective_user

    if not context.args:
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            "❌ Please provide a Game ID.\n\nExample: `/join ABC123`",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    game_id = context.args[0].upper().strip()

    # Validate game ID format
    if len(game_id) != GAME_CONFIG.GAME_ID_LENGTH or not all(c in GAME_CONFIG.GAME_ID_CHARS for c in game_id):
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            f"❌ Invalid Game ID format: `{game_id}`\n\nGame IDs are 6 letters/numbers. Example: `ABC123`",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    session = games.get(game_id)

    if session is None:
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            f"❌ No active game found with ID `{game_id}`.\n\nCheck the ID and try again, or create a new game with `/newgame`",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    if session.is_ended():
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            f"❌ Game `{game_id}` has already ended.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    if session.is_active():
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            f"❌ Game `{game_id}` has already started. You cannot join mid-game.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    if user.id in session.players:
        keyboard = [
            ["📋 /status", "⏹️ /endgame"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            f"ℹ️ You've already joined game `{game_id}`. Waiting for the host to start.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    # Add player to game
    if session.add_player(user.id, user.first_name):
        db.save_game(session)

        keyboard = [
            ["📋 /status", "🎮 /newgame"],
            ["❓ /help"],
        ]
        player_list = "\n".join(f"• {p.name}" for p in session.players.values())
        await update.message.reply_text(
            f"✅ You joined game `{game_id}`!\n\n"
            f"*Current players ({len(session.players)}):*\n{player_list}\n\n"
            f"⏳ Waiting for the host to start the game.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )

        # Notify the host about new player
        if user.id != session.host_id:
            try:
                host_keyboard = [
                    ["📋 /status", "▶️ /startgame"],
                    ["⏹️ /endgame", "❓ /help"],
                ]
                await context.bot.send_message(
                    chat_id=session.host_id,
                    text=f"👤 *{user.first_name}* joined your game `{game_id}`.\n\nPlayers now: {len(session.players)}",
                    parse_mode="Markdown",
                    reply_markup=ReplyKeyboardMarkup(host_keyboard, one_time_keyboard=True, resize_keyboard=True),
                )
            except Exception as e:
                logger.warning(f"Could not notify host: {e}")

        logger.info("Player %s (%d) joined game %s", user.first_name, user.id, game_id)
    else:
        logger.warning("Failed to add player %d to game %s", user.id, game_id)


async def cmd_startgame(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Start a game (host only)."""
    user = update.effective_user

    # Find host's active game
    session = None
    for g in games.values():
        if g.host_id == user.id and not g.is_ended():
            session = g
            break

    if session is None:
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            "❌ You don't have an active game. Use `/newgame` to create one.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    if session.is_active():
        keyboard = [
            ["📋 /status", "⏹️ /endgame"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            "ℹ️ The game has already started.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    if len(session.players) < GAME_CONFIG.MIN_PLAYERS:
        keyboard = [
            ["📋 /status", "👥 Share game ID"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            f"❌ You need at least {GAME_CONFIG.MIN_PLAYERS} players to start.\n\n"
            f"Current: {len(session.players)}\n"
            f"Share your Game ID: `{session.game_id}`",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    # Assign unique secrets
    player_ids = list(session.players.keys())
    available_secrets = PLAYER_SECRETS.copy()
    random.shuffle(available_secrets)

    for i, pid in enumerate(player_ids):
        secret = available_secrets[i % len(available_secrets)]
        session.secrets[pid] = secret
        session.players[pid].secret = secret

    session.status = GameStatus.ACTIVE
    from datetime import datetime
    session.started_at = datetime.utcnow()
    db.save_game(session)

    # Send each player their secret via DM with keyboard
    for pid, player in session.players.items():
        secret = session.secrets[pid]
        try:
            keyboard = [
                ["📋 /status"],
                ["❓ /help"],
            ]
            reply_markup = ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True)
            await context.bot.send_message(
                chat_id=pid,
                text=(
                    f"🔍 *The game has begun.*\n\n"
                    f"*Your secret:*\n_{secret}_\n\n"
                    f"Messages will find you as the story unfolds. "
                    f"Respond to them by contacting the other players directly.\n\n"
                    f"⏱️ Game duration: {GAME_CONFIG.GAME_DURATION_MINUTES} minutes\n"
                    f"Alice is missing. Find out what happened."
                ),
                parse_mode="Markdown",
                reply_markup=reply_markup,
            )
        except Exception as e:
            logger.warning(
                "Could not send secret to player %s (%d): %s", player.name, pid, e
            )

    # Import and start trigger loops
    import asyncio
    from utils import player_trigger_loop

    for pid in session.players:
        task = asyncio.create_task(player_trigger_loop(session, pid, context.bot))
        session.player_tasks[pid] = task

    keyboard = [
        ["📋 /status"],
        ["⏹️ /endgame", "❓ /help"],
    ]
    await update.message.reply_text(
        f"🎬 *Game started!* {len(session.players)} players have received their secrets.\n\n"
        f"Use `/endgame` to end the session.",
        parse_mode="Markdown",
        reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
    )
    logger.info("Game %s started with %d players", session.game_id, len(session.players))


async def cmd_endgame(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """End a game (host only)."""
    user = update.effective_user

    # Find host's active game
    session = None
    for g in games.values():
        if g.host_id == user.id and not g.is_ended():
            session = g
            break

    if session is None:
        keyboard = [
            ["🎮 /newgame", "📋 /status"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            "❌ You don't have an active game.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    await _end_game(session, context.bot)

    elapsed = int(session.elapsed_minutes())
    keyboard = [
        ["🎮 /newgame", "📋 /status"],
        ["❓ /help"],
    ]
    await update.message.reply_text(
        f"🏁 *Game ended.*\n\n"
        f"Duration: {elapsed} minutes\n"
        f"Players: {len(session.players)}\n\n"
        f"Thank you for playing. What really happened to Alice?",
        parse_mode="Markdown",
        reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
    )
    logger.info("Game %s ended by host", session.game_id)


async def cmd_status(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Show current game status."""
    user = update.effective_user

    # Find any game the user is part of
    session = None
    for g in games.values():
        if user.id in g.players and not g.is_ended():
            session = g
            break

    if session is None:
        keyboard = [
            ["🎮 /newgame", "👥 /join"],
            ["❓ /help"],
        ]
        await update.message.reply_text(
            "ℹ️ You are not in any active game.\n\nUse `/newgame` to create one or `/join <game_id>` to join.",
            parse_mode="Markdown",
            reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
        )
        return

    player_list = "\n".join(f"• {p.name}" for p in session.players.values())
    
    if session.status == GameStatus.LOBBY:
        status_str = "🔵 Lobby — Waiting to start"
    elif session.is_active():
        status_str = f"🟢 Active — {session.phase().value.upper()} phase"
    else:
        status_str = "⚫ Ended"

    elapsed = int(session.elapsed_minutes()) if session.is_active() else 0

    msg = (
        f"📋 *Game Status*\n\n"
        f"ID: `{session.game_id}`\n"
        f"Status: {status_str}\n"
        f"{'⏱️ Elapsed: ' + str(elapsed) + ' min' if session.is_active() else ''}\n\n"
        f"*Players ({len(session.players)}):*\n{player_list}"
    )

    keyboard = [
        ["🎮 /newgame", "👥 /join"],
        ["❓ /help"],
    ]
    if user.id == session.host_id and session.status == GameStatus.LOBBY:
        keyboard.insert(1, ["▶️ /startgame", "⏹️ /endgame"])
    
    await update.message.reply_text(
        msg.strip(),
        parse_mode="Markdown",
        reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
    )
    logger.info("Status requested by user %d in game %s", user.id, session.game_id)


async def _end_game(session: GameSession, bot) -> None:
    """End a game and notify all players."""
    session.status = GameStatus.ENDED
    from datetime import datetime
    session.ended_at = datetime.utcnow()
    db.save_game(session)

    # Cancel all player tasks
    for task in session.player_tasks.values():
        if not task.done():
            task.cancel()

    # Notify all players
    for pid, player in session.players.items():
        try:
            keyboard = [
                ["🎮 /newgame", "📋 /status"],
                ["❓ /help"],
            ]
            await bot.send_message(
                chat_id=pid,
                text=(
                    "🏁 *The game has ended.*\n\n"
                    "The host has closed this session. "
                    "Whatever you know about Alice — it's time to share it."
                ),
                parse_mode="Markdown",
                reply_markup=ReplyKeyboardMarkup(keyboard, one_time_keyboard=True, resize_keyboard=True),
            )
        except Exception as e:
            logger.warning("Could not notify player %s (%d): %s", player.name, pid, e)

    logger.info("Game %s ended", session.game_id)
