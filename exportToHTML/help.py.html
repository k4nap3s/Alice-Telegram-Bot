""" 
handlers/help.py — Help and information commands. 
"""

import logging
from telegram import Update, ReplyKeyboardMarkup
from telegram.ext import ContextTypes

logger = logging.getLogger(__name__)


async def cmd_help(update: Update, context: ContextTypes.DEFAULT_TYPE) -> None:
    """Send help message with reply keyboard."""
    keyboard = [
        ["🎮 /newgame", "👥 /join"],
        ["▶️ /startgame", "⏹️ /endgame"],
        ["📋 /status", "❓ /help"],
    ]
    reply_markup = ReplyKeyboardMarkup(
        keyboard, one_time_keyboard=True, resize_keyboard=True
    )

    text = (
        "🕵️ *Alice Is Missing — Telegram Edition*\n\n"
        "*Commands:*\n"
        "`/newgame` — Create a new game session\n"
        "`/join <game_id>` — Join an existing game\n"
        "`/startgame` — Begin the game (host only)\n"
        "`/endgame` — End the game (host only)\n"
        "`/status` — Show game status\n"
        "`/help` — Show this message\n\n"
        "Once the game starts, you will receive your secret and personal trigger cards "
        "via private message every few minutes. "
        "Play in silence — all communication should happen through your private chats with other players."
    )

    await update.message.reply_text(
        text, parse_mode="Markdown", reply_markup=reply_markup
    )
    logger.info("Help command sent to user %s", update.effective_user.id)
